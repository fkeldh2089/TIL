# 복잡한 조립라인 2

```python
import sys


K, N = map(int, input().split())

times = []
for p in range(N):
    ho = list(map(int, input().split()))
    times.append(ho)

if N == 1:
    print(min(times[0]))

else:
    visited = [[0]*K for _ in range(N)]
    for p1 in range(N-1):
        mn1 = 1000000000000
        for p2 in range(K):
            a = visited[p1-1][p2] + times[p1][p2]
            visited[p1][p2] = a
            a += times[p1][-1]
            if mn1>a:
                mn1 = a
        for p3 in range(K):
            if visited[p1][p3] > mn1:
                visited[p1][p3] = mn1
    ret = 1000000000000
    for p1 in range(K):
        tmp = visited[N-2][p1] + times[N-1][p1]
        if tmp < ret:
            ret = tmp
    print(ret)
```
