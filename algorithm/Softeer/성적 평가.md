# 성적 평가

````python
import sys

N = int(input())
tests = []
testRes = [[0]*4002 for _ in range(4)] 
for p in range(3):
    test = list(map(int, input().split()))
    tests.append(test)

tot = []
for p1 in range(N):
    tmp = 0
    for p2 in range(3):
        tmp += tests[p2][p1]
    tot.append(tmp)
tests.append(tot)

for q in range(4):
    for p in range(N):
        testRes[q][tests[q][p]] += 1

for p1 in range(4):
    for p2 in range(4001):
        testRes[p1][p2+1] += testRes[p1][p2]

for p1 in range(4):
    # print(tests[p1])
    for p2 in range(N):
        print(N-testRes[p1][tests[p1][p2]]+1, end=" ")
    print()
````
