def solution(m, musicinfos):
    answer = ''
    # 전처리
    mn = 0
    mn2 = 24*60
    for q in range(len(musicinfos)):
        st, et, title, code = musicinfos[q].split(',')
        sht, smt = st.split(':')
        eht, emt = et.split(':')
        playtime = (int(eht)-int(sht))*60 + (int(emt)-int(smt))  # 시간
        
        f = 0
        f2 = 0
        i = 0  # 노래 길이 카운트
        tmpi = 0  # 노래 코드 인덱스
        ii = 0 # 해당 노래가 맞는지 체크
        while i < playtime+1:
            if i == playtime and code[tmpi] != '#':
                break
            if code[tmpi] == m[0]:
                if f == 1:
                    flag = tmpi
                    flagi = i
                    f2 = 1
            if code[tmpi] == m[ii]:
                ii += 1
                f = 1
            else:
                if f2 == 1:
                    tmpi = flag-1
                    i = flagi-1
                    ii = 0
                    f = 0
                    f2 = 0
                else:
                    ii = 0
                    f = 0
            if code[tmpi] == '#':
                tmpi += 1
                tmpi %= len(code)
            else:
                tmpi += 1
                i += 1
                tmpi %= len(code)
            if ii == len(m):
                if code[tmpi] == '#':
                    ii = 0
                else:
                    if mn < playtime:
                        mn = playtime
                        answer = title
                        break
                    break
    if answer == '':
        answer = "(None)"
    return answer
