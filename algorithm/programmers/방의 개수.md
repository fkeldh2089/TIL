from collections import defaultdict

def solution(arrows):
    # 평행으로 두번 반복 되지 않고 교차하면,
    answer = 0
    dr = [-1, -1, 0, 1, 1, 1, 0, -1]
    dc = [0, 1, 1, 1, 0, -1, -1, -1]
    r, c,  = 0, 0
    graph = defaultdict(lambda: defaultdict(lambda: [0, 0, 0, 0, 0, 0, 0, 0]))
    for arrow in arrows:
        f1, f2 = 1, 0  # f1
        if graph[r][c][arrow]:
            f1 = 0
        else:
            graph[r][c][arrow] = 1
        r += dr[arrow]
        c += dc[arrow]
        if graph[r][c] != [0, 0, 0, 0, 0, 0, 0, 0]:
            f2 = 1
        if graph[r][c][arrow-4]:
            f1 = 0
        else:
            graph[r][c][arrow-4] = 1
        if f1 and f2:
            answer += 1
        if f1 and arrow in [1, 3, 5, 7]:
            nr, nc = r+dr[arrow-3], c+dc[arrow-3]
            if graph[nr][nc][arrow-6] == 1:
                answer += 1
    return answer
